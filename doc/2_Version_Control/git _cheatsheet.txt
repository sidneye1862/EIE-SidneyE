ðŸ”¹ git commit
----------------
# Create a commit with a message

# Stage and commit tracked files in one step

ðŸ”¹ git branch
----------------
# List all local branches
git branch

# Create a new branch
git branch new-branch

# Delete a branch

ðŸ”¹ git checkout
---------------------
# Switch to another branch
git checkout branch-name

# Create and switch to a new branch
git checkout -b new-branch

# Discard local changes in a file
git checkout -- filename


ðŸ”¹ git pull
---------------
# Fetch and merge changes from the remote branch

ðŸ”¹ git push
---------------
# Push current branch to remote
git push origin branch-name

# Push all local branches
git push --all origin


ðŸ”¹ git merge
----------------
# Merge a branch into the current branch
git merge branch-name

# (Resolve conflicts manually if prompted)


ðŸ”¹ git rebase
-----------------
# Rebase your feature branch onto main

git rebase main

# Interactive rebase (e.g., squash commits)
git rebase -i HEAD~3


ðŸ”¹ git fetch
----------------
# Download latest changes from remote (no merge)
git fetch origin

# Fetch a specific branch
